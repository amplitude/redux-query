const serveRequest = (url, method, { body, headers }, callback) => {
    if (url === '/api/echo' && method.toUpperCase() === 'POST') {
        // Respond with the request body's "name" field after an artificial timeout
        setTimeout(function() {
            callback(200, body.name);
        }, 1000);
    } else {
        callback(404);
    }
};

export default serveRequest;
