const serveRequest = (url, method, { body, headers }, callback) => {
    if (url === '/api/hello' && method.toUpperCase() === 'GET') {
        var response = { message: 'World' };

        // Respond after an artificial timeout
        setTimeout(function() {
            callback(200, response);
        }, 1000);
    } else {
        callback(404);
    }
};

export default serveRequest;
